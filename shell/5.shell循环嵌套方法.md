## shell循环嵌套使用方法

### 1. 双重循环
> echo -n "aaa" 输出内容后不换行
> echo $Char 不行 不能直接执行echo * 只能写成echo "*"
1. 例子
```shell
# 例1: 使用特殊符号打印三角形。要求整个程序要有交互，运行时可以自动输入打印的行数和用于描绘三角形的特殊符号。
read -p "Please Enter the line number: " Line
read -p "Please Enter the char number: " Char
echo "$Line $Char"
Vline=1
while ((Vline<=Line))
do
  temp=1
  while ((temp<=Vline))
  do
    # 不换行，同行输出
    echo -n "$Char"
    temp=`expr $temp + 1`
  done
  # 换行
  echo
  Vline=`expr $Vline + 1`
done

# 结果2
# Please Enter the line number: 5
# Please Enter the char number: *
# 5 *
# *
# **
# ***
# ****
# *****

# 例2:
# 想要输出
#   *
#  ***
# *****
# 方法1: 直接输出
echo "  *  "
echo " *** "
echo "*****"
# 方法2: cat 和 文件结束符
cat <<EOF
  *
 ***
*****
EOF
# 方法3: 编码方式
read -p "Please Enter a number:" Line
for ((i=1; i<=Line; i++))
do
  for((j=Line-i; j>0; j--))
  do
    echo -n " "
  done
  for ((h=1; h<=((2*i-1)); h++))
  do
  echo -n "*"
  done
  echo
done

```
