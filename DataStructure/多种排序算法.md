## 多种排序算法
### 快排
```js
const quickSort = (arr) => {
  const sort = (arr, left = 0, right = arr.length - 1) => {
    if (left >= right) {
      return
    }
    let i = left;
    let j = right;
    // 哨兵
    const temp = arr[left];
    while (i < j) {
      while (i < j && arr[j] > temp) {
        j--;
      }
      while (i < j && arr[i] <= temp) {
        i++;
      }
      // 快排每次确定一个元素的真实位置
      if (i < j) {
        [arr[i], arr[j]] = [arr[j], arr[i]];
      }
    }
    // 哨兵与下标i的元素交换
    [arr[left], arr[i]] = [arr[i], arr[left]];
    // 递归快排元素左右两边的数组
    sort(arr, left, i - 1);
    sort(arr, i + 1, right);
  }
  sort(arr);
  return arr;
}
var arr = [3, 6, 12, 7, 2, 7, 3];
console.log(quickSort(arr));
```
