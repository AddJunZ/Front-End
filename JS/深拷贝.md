## 深拷贝

### 1. 简单拷贝
缺点1：没法克隆函数，错误复制正则和symbol
```js
const clone = (obj) => {
  return JSON.parse(JSON.stringify(obj));
}

// test
let a = {
	reg: /^asd$/,
	fn: function t(x) {
		return x;
	},
	syb: Symbol('foo'),
	asd: 'asd'
}
const DeepClone = (obj) => {
	return JSON.parse(JSON.stringify(obj));
}
console.log(DeepClone(a));
// { reg: {}, asd: 'asd' }

```
缺点2：对对象的引用有问题
```js
let obj = { value: 'value' }; 
let info = { name: 'AddJunZ' };
obj.v1 = info;
obj.v2 = info;// 两个属性指向同一个对象
let cp = JSON.parse(JSON.stringify(obj)); 
obj.v1.name = 'changedName'; 
cp.v1.name  = 'changedName';// 复制后的对象的两个值并不都指向同一个对象
console.log(obj,cp);

/*
{
  value: 'value',
  v1: { name: 'changedName' },
  v2: { name: 'changedName' }
} {
  value: 'value',
  v1: { name: 'changedName' },
  v2: { name: 'AddJunZ' }
}
*/
```

### 2. 递归遍历
```js
const DeepClone = (target) => {
	let result;
	if(typeof target === 'object'){
		// 如果是数组
		if(Array.isArray(target)){
			result = [];
			for(let i in target){
				result.push(DeepClone(target[i]));
			}
		}else if(target === null || target instanceof RegExp || target instanceof Date){
			result = target;
		}else{
			result = {};
			for(let prop in target){
				result[prop] = DeepClone(target[prop]);
			}
		}
	}else{
		result = target;
	}
	return result;
}
```
