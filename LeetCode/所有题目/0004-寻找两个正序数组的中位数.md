### 一般做法
```js
var findMedianSortedArrays = function(nums1, nums2) {
  const arr = [...nums1, ...nums2].sort((a,b)=> a-b);
  const len = arr.length;
  if(len == 0)return 0;
  if(len == 1)return arr[0];
  if(len % 2 == 0){
    const mid = len / 2;
    return (arr[mid - 1] + arr[mid]) / 2;
  } else {
    return arr[Math.floor(len / 2)]
  }
};
```
